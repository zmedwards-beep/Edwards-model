const edwardsDictionary = {
    "0,0,0,0,0": { name: "Empty Void", desc: "Baseline depletion." },
    "1,1,1,0,0": { name: "Anger/Rage (Lövheim)", desc: "The classic 3D high-energy aggression." },
    "1,1,1,1,0": { name: "Heroic Radiance", desc: "Aggression redirected into social protection (The O-Filter)." },
    "1,1,1,1,1": { name: "The Edwards Peak", desc: "Total neurochemical saturation and unity." },
    "0,0,1,1,0": { name: "Grief", desc: "Arousal and Social Bond without Reward or Stability." },
    "1,0,1,0,1": { name: "Indomitable Resilience", desc: "Stability and focus buffered by Endorphins." }
    // Full 32-node dictionary would be populated here
};

let ctx = document.getElementById('edwardsRadar').getContext('2d');
let radar = new Chart(ctx, {
    type: 'radar',
    data: {
        labels: ['Serotonin', 'Dopamine', 'Noradrenaline', 'Oxytocin', 'Endorphins'],
        datasets: [{
            data: [0, 0, 0, 0, 0],
            backgroundColor: 'rgba(78, 204, 163, 0.2)',
            borderColor: '#4ecca3',
            borderWidth: 3
        }]
    },
    options: {
        scales: { r: { min: 0, max: 1, ticks: { display: false }, grid: { color: '#0f3460' } } },
        plugins: { legend: { display: false } }
    }
});

function update() {
    const vals = [
        document.getElementById('S').value,
        document.getElementById('D').value,
        document.getElementById('N').value,
        document.getElementById('O').value,
        document.getElementById('E').value
    ];
    
    radar.data.datasets[0].data = vals;
    radar.update();

    const key = vals.join(',');
    const node = edwardsDictionary[key] || { name: "Hybrid State", desc: "A complex mix of neurochemical influences." };
    
    document.getElementById('node-id').innerText = `Current Vector: [${key}]`;
    document.getElementById('node-display').innerText = node.name;
    document.getElementById('node-desc').innerText = node.desc;
}
