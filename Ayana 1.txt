<script>
const nodes = {
    "0,0,0,0,0": { name: "Empty Void", desc: "Total neurochemical depletion and apathy." },
    "1,1,1,1,1": { name: "The Edwards Peak", desc: "Total neurochemical saturation; transcendental unity." },
    "0,0,1,1,0": { name: "Grief / Anguish", desc: "High-arousal social loss and distress." },
    "1,1,1,0,1": { name: "Flow State", desc: "Peak performance and internal focus; pain-buffering active." },
    "1,0,1,1,1": { name: "Universal Sublime", desc: "Total peace and connection under high arousal." },
    "0,0,1,0,0": { name: "Paranoid Terror", desc: "Pure high-alert fear in total isolation." },
    "1,1,1,1,0": { name: "Heroic Radiance", desc: "Selfless peak performance; social drive." },
    "1,0,0,1,1": { name: "Domestic Tranquility", desc: "Deep safety, bonding, and physical comfort." }
    // More nodes can be added here from the 32-node dictionary
};

let ctx = document.getElementById('radarChart').getContext('2d');
let radarChart = new Chart(ctx, {
    type: 'radar',
    data: {
        labels: ['Serotonin', 'Dopamine', 'Noradrenaline', 'Oxytocin', 'Endorphins'],
        datasets: [{
            label: 'Neurochemical Levels',
            data: [0, 0, 0, 0, 0],
            backgroundColor: 'rgba(78, 204, 163, 0.2)',
            borderColor: '#4ecca3',
            pointBackgroundColor: '#4ecca3'
        }]
    },
    options: {
        scales: { r: { min: 0, max: 1, ticks: { display: false }, grid: { color: '#0f3460' }, angleLines: { color: '#0f3460' } } },
        plugins: { legend: { display: false } }
    }
});

function updateModel() {
    const s = document.getElementById('S').value;
    const d = document.getElementById('D').value;
    const n = document.getElementById('N').value;
    const o = document.getElementById('O').value;
    const e = document.getElementById('E').value;
    
    radarChart.data.datasets[0].data = [s, d, n, o, e];
    radarChart.update();

    const key = `${s},${d},${n},${o},${e}`;
    const node = nodes[key] || { name: "Hybrid State", desc: "A transitional emotional state between primary nodes." };
    
    document.getElementById('emotion-name').innerText = node.name;
    document.getElementById('emotion-desc').innerText = node.desc;
}
