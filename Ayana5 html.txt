<script>
const edwardsDictionary = {
    "0,0,0,0,0": { name: "Empty Void", desc: "Baseline depletion; clinical apathy." },
    "1,0,0,0,0": { name: "Stoic Detachment", desc: "Internal stability without external drive or bond." },
    "0,0,1,0,0": { name: "Paranoid Terror", desc: "High arousal (N) with zero social filter (O)." },
    "0,0,1,1,0": { name: "Grief / Anguish", desc: "High arousal social longing; acute social pain." },
    "1,0,1,1,1": { name: "Universal Sublime", desc: "Awe, connection, and equanimity under high arousal." },
    "1,1,1,1,1": { name: "The Edwards Peak", desc: "Total neurochemical saturation; transcendental unity." },
    "1,0,1,0,1": { name: "Indomitable Resilience", desc: "Stability and focus buffered by endorphins." },
    "1,1,1,1,0": { name: "Heroic Radiance", desc: "Drive and arousal filtered through social bonding." },
    "0,1,1,0,1": { name: "Predatory Flow", desc: "Reward-seeking arousal without empathy, buffered by endorphins." }
};

let ghostData = null;
let ctx = document.getElementById('edwardsRadar').getContext('2d');

let radar = new Chart(ctx, {
    type: 'radar',
    data: {
        labels: ['Serotonin', 'Dopamine', 'Noradrenaline', 'Oxytocin', 'Endorphins'],
        datasets: [{
            label: 'Current State',
            data: [0, 0, 0, 0, 0],
            backgroundColor: 'rgba(78, 204, 163, 0.3)',
            borderColor: '#4ecca3',
            borderWidth: 3,
            zIndex: 2
        }, {
            label: 'Pinned Target',
            data: null,
            backgroundColor: 'rgba(255, 255, 255, 0.05)',
            borderColor: 'rgba(255, 255, 255, 0.3)',
            borderDash: [5, 5],
            borderWidth: 2,
            hidden: true,
            zIndex: 1
        }]
    },
    options: {
        scales: { r: { min: 0, max: 1, ticks: { display: false }, grid: { color: '#0f3460' }, angleLines: { color: '#0f3460' } } },
        plugins: { legend: { display: false } }
    }
});

function update() {
    const s = document.getElementById('S').value;
    const d = document.getElementById('D').value;
    const n = document.getElementById('N').value;
    const o = document.getElementById('O').value;
    const e = document.getElementById('E').value;
    const current = [s, d, n, o, e];
    
    radar.data.datasets[0].data = current;
    radar.update();

    const key = current.join(',');
    const node = edwardsDictionary[key] || { name: "Hybrid State", desc: "A complex neurochemical transition." };
    document.getElementById('node-display').innerText = node.name;
    document.getElementById('node-desc').innerText = node.desc;

    if (ghostData) {
        calculateDelta(current, ghostData);
    }
}

function saveGhost() {
    ghostData = [...radar.data.datasets[0].data];
    radar.data.datasets[1].data = ghostData;
    radar.data.datasets[1].hidden = false;
    radar.update();
    document.getElementById('delta-report').style.display = 'block';
    update();
}

function resetGhost() {
    ghostData = null;
    radar.data.datasets[1].hidden = true;
    radar.update();
    document.getElementById('delta-report').style.display = 'none';
}

function calculateDelta(current, target) {
    const labels = ['Serotonin', 'Dopamine', 'Noradrenaline', 'Oxytocin', 'Endorphins'];
    let shifts = [];
    for (let i = 0; i < 5; i++) {
        let diff = target[i] - current[i];
        if (diff > 0) shifts.push(`+ ${labels[i]}`);
        else if (diff < 0) shifts.push(`- ${labels[i]}`);
    }
    document.getElementById('delta-text').innerText = shifts.length ? shifts.join(' | ') : "States are identical.";
}
